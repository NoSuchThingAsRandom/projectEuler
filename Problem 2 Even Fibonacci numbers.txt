    public static void main(String[] args) {
        int a=1;
        int b=1;
        int c=a+b;
        int sum=2;
        while(c<4000000){
            a=b;
            b=c;
            c=a+b;
            if(c%2==0){
                sum=sum+c;
            }
        }
        System.out.println("Total="+sum);
    }
